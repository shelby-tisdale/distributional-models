demog_long <- demog |>
  select(src_subject_id, starts_with("demo_race_a_p___"), starts_with("demo_ethn")) |>
  # pivot race variables
  pivot_longer(cols = starts_with("demo_race_a_p___"),
               names_to = "race",
               values_to = "is_race",
               values_drop_na = TRUE,
               names_prefix = "demo_race_a_p___") |>
  filter(is_race != 0) |>
  mutate(race = fct_collapse(
    race,
    "White" = 10,
    "Black" = 11,
    "Native American" = 12,
    "Alaskan Native" = 13,
    "Native Hawaiian" = 14,
    "Guamanian" = 15,
    "Samoan" = 16,
    "Other Pacific Islander" = 17,
    "Asian Indian" = 18,
    "Chinese" = 19,
    "Filipino" = 20,
    "Japanese" = 21,
    "Korean" = 22,
    "Vietnamese" = 23,
    "Other Asian Race" = 24,
    "Other Race" = 25,
    "NA" = c(77, 99, 0)
  )) |>
  distinct() |>
  select(-is_race) |>
  group_by(src_subject_id) |>
  mutate(order = row_number()) |> 
  
  #clean hispanic origin variable 
  mutate(hispanic_origin = case_when( 
    demo_ethn2_v2 == 10 ~ "Puerto Rican", 
    demo_ethn2_v2 == 12 ~ "Dominican",
    demo_ethn2_v2 == 13 ~ "Mexican", 
    demo_ethn2_v2 == 14 ~ "Mexican American", 
    demo_ethn2_v2 == 15 ~ "Chicano", 
    demo_ethn2_v2 == 18 ~ "Cuban", 
    demo_ethn2_v2 == 19 ~ "Cuban American", 
    demo_ethn2_v2 == 20 ~ "Central or South American", 
    demo_ethn2_v2 == 40 ~ "Other Latin American origin", 
    demo_ethn2_v2 == 41 ~ "Other Hispanic", 
    demo_ethn2_v2 %in% c(777, 999) ~ "NA"
    ))
